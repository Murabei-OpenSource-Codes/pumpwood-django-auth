 30def get_user_permissions(user_id: int, route_name: str = None):
 31    query_results = None
 32    if route_name is not None:
 33        query = """
 34            SELECT *
 35            FROM public.api_permission__list_all_permissions
 36            WHERE user_id = %(user_id)s
 37              AND route_name = %(route_name)s
 38        """
 39        query_results = pd.read_sql(
 40            query, con=connection, params={
 41                "user_id": user_id,
 42                "route_name": route_name})
 43    else:
 44        query = """
 45            SELECT *
 46            FROM public.api_permission__list_all_permissions
 47            WHERE user_id = %(user_id)s
 48        """
 49        query_results = pd.read_sql(
 50            query, con=connection, params={
 51                "user_id": user_id
 52            })
 53
 54    # Custom action policy
 55    is_custom_action = query_results["can_run_actions"] == 'custom'
 56    custom_action = query_results[is_custom_action]
 57    unique_policy_id = custom_action["policy_id"].dropna().unique().tolist()
 58    action_policy = pd.DataFrame(
 59        PumpwoodPermissionPolicyAction.objects
 60        .filter(policy_id__in=unique_policy_id)
 61        .values('policy_id', 'action', 'permission'),
 62        columns=['policy_id', 'action', 'permission'])
 63    custom_action = custom_action[[
 64        "policy_id", 'user_id', 'group_id', "priority",
 65        "route_id", "route_name"]].merge(
 66        action_policy, on="policy_id", validate="1:m")
 67    return {
 68        "permission_policy": query_results.replace({np.nan: None}),
 69        "permission_custom_action": custom_action.replace({np.nan: None})}

 72@api_view(['GET'])
 73def view__list_self_permissions(request):
 74    """Get kong routes."""
 75    user = request.user
 76    route_name = request.GET.get('route_name')
 77    query_results = get_user_permissions(
 78        user_id=user.id, route_name=route_name)
 79    return Response(query_results)
